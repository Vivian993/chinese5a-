    // === 一鍵批次輸入生字 ===
    function batchInputCharacters(lessonId) {
      const lesson = appData.lessons.find(l => l.id === lessonId);
      const inputArea = document.getElementById(`batch-input-${lessonId}`);
      if (!lesson || !inputArea) return;

      const inputText = inputArea.value.trim();
      if (!inputText) return;

      // 儲存輸入文本
      lesson.batchInput = inputText;

      const lines = inputText.split('\n');
      lesson.characters = []; // 清空現有生字

      lines.forEach((line, index) => {
        const parts = line.split(',');
        if (parts.length < 4) return; // 格式不符

        const charObj = {
          id: index + 1,
          char: parts[0].trim(),
          zhuyin: parts[1].trim(),
          radical: parts[2].trim(),
          note: parts.slice(3).join('、').trim(), // 圈詞1,圈詞2 -> 圈詞1、圈詞2
          terms: parts.slice(3).join('、').trim(),
          tingxieVisible: true
        };
        lesson.characters.push(charObj);
      });

      renderLessonCharList(lessonId); // 重新渲染此課生字列表
    }

    async function searchDict() {
      const query = document.getElementById('dict-query-input').value.trim();
      if (!query) return;

      const resultDisplay = document.getElementById('dict-result-content');
      resultDisplay.innerText = "查詢中...";

      try {
        const response = await fetch(`https://www.moedict.tw/a/${encodeURIComponent(query)}.json`);
        if (!response.ok) throw new Error('Not found');
        const data = await response.json();
       
        // 簡單格式化輸出
        let output = `<strong>【${data.title}】</strong><br>`;
        if(data.heteronyms && data.heteronyms.length > 0) {
          data.heteronyms.forEach(h => {
            if(h.bopomofo) output += `<span style="color:var(--macaron-pink)">[${h.bopomofo}]</span><br>`;
            if(h.definitions) {
              h.definitions.forEach(def => {
                output += def.def.replace(/`|~/g, '') + "<br>";
              });
            }
          });
        }
        resultDisplay.innerHTML = output;
      } catch (error) {
        resultDisplay.innerText = "查無此字或網路失敗，請試萌典官網。";
      }
    }
